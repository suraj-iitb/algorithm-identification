#include<bits/stdc++.h>

int sort(int *a,int n){
  int count=0;
  bool j=true;
  while(j){
    j=false;
    for(int i=1;i<n;i++){
    if(*(a+i-1)>*(a+i)){j=true; swap(*(a+i-1),*(a+i)); count++;}
  }}


  return count;
}
