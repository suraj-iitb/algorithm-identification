void insertionSort(int A[], int N) {
	a(A, N);


	for (int i = 1; i < N; i++) {
		int v = A[i],
			j = i - 1;
		while (j >= 0 && A[j] > v) {
			A[j + 1] = A[j];
			j--;
		}
		A[j + 1] = v;
		a(A, N);
	}
}
