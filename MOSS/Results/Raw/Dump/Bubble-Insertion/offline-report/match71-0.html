int bubble_sort (vector<int> & a )
{
	int n = a.size();
	int res = 0;
	rep (i, n ){


		for (int j = n-1; j >= i + 1; j-- ){
			if (a[j] < a[j-1] ){
				swap (a[j], a[j-1] );
				res++;
			} // end if
		} // end for
	} // end rep

	return res;
}
