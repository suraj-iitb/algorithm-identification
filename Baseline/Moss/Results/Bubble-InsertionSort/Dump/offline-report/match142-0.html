void BubbleSort::solve( int n, int a[] )
{
  int cnt = 0;
  


  for( int i = 0; i < n; i++ )
    {
      for( int j = n - 1; j > i;j-- )
        {
          if( a[j] < a[j-1] )
            {
              swap( a[j], a[j-1] );
              cnt++;
            }
        }
    }

  for( int i = 1; i < n; i++ )
    {
      cout << " " << a[i] ;
    }
  cout << endl;
  cout << cnt << endl;
}
