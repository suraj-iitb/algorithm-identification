const int MAX_SIZE = 100;

int d[MAX_SIZE];
int ans = 0;

void bubble_sort(int n){
	bool f = true;
	while(f){
		f = false;
		for(int j = n -1; j > 0; j--){
			if(d[j] < d[j - 1]){
				swap(d[j], d[j - 1]);
				f = true;
				ans++;
			}
		}
	}
}
