void insertionsort(int A[],int N)
{
    int j,i,v;


  for(int i=1;i<N;i++)
  {
      v=A[i];
      j=i-1;
  while(j>=0&&A[j]>v)
  {
      A[j+1]=A[j];
      j--;
  }
      A[j+1]=v;
      trace(A,N);
  }
}
