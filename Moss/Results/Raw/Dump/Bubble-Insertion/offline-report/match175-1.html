int n,A[100];

int Bubblesort(int *p,int size){
	int count = 0;
	for(int i = 0;i < size-1;i++){


		for(int j = size-1;j >= i+1;j--){
			if(A[j] < A[j-1]){
				swap(A[j],A[j-1]);
				count++;
			}
		}
	}
	return count;
}
