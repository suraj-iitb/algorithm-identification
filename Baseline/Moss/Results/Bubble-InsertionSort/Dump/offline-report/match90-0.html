int main(){
	int n;
	int hoge[100];
	int sum=0;
	cin >> n;
	for(int i=0; i<n; ++i){
		cin >> hoge[i];
	}
	for(int i=0; i<n; ++i){


		for(int j=n-1; j>i ;--j){
			if(hoge[j] < hoge[j-1]){
				int tmp = hoge[j];
				hoge[j] = hoge[j-1];
				hoge[j-1] = tmp;
				++sum;
			}
		}
	}
	for(int i=0; i<n; ++i){
		if(i+1 != n)
			cout << hoge[i] << " ";
		else 
			cout << hoge[i] << endl;
	}
	cout << sum << endl;
}
