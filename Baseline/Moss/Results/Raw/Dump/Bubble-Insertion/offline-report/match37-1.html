class Insertionsort
{
 public:
  void solve( int n, int a[] );
};

void Insertionsort::solve( int n, int a[] )
{
  int key;
  


  for( int i = 0; i < n - 1; i++ )
    {
      cout << a[i] << " ";
    }
  cout << a[n-1] << endl;
      
  for( int j = 1; j < n; j++ )
    {
      key = a[j];
      int k = j - 1;


      while((k >= 0) && (a[k] > key))
        {
          a[k+1] = a[k];
          k--;
        }
      a[k+1] = key;
        
      for( int i = 0; i < n - 1; i++ )
        {
          cout << a[i] << " ";
        }
      cout << a[n-1] << endl;
    }
  
}
